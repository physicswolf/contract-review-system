### 1.3 新端点 — 发起审核时写入合同记录

**文件**：`api/contracts.py`

新增 `POST /api/contracts/audit`：

```python
@router.post("/audit")
async def start_audit_from_upload(payload: dict[str, Any]) -> JSONResponse:
    """发起审核。此时才写入 contracts 表。"""
    file_id = str(payload.get("fileId") or "")
    role = str(payload.get("role") or "甲方")
    point_ids = [int(pid) for pid in payload.get("points") or []]
    contract_type = str(payload.get("contractType") or "未分类")

    if not file_id:
        return error_response(422, "FILE_ID_REQUIRED", "请提供文件 ID")
    if not point_ids:
        return error_response(422, "POINTS_REQUIRED", "请至少选择一个审核点")

    settings = get_settings()

    # 1. 从解析 JSON 提取合同元信息
    try:
        meta = extract_contract_meta(file_id, settings)
    except Exception:
        return error_response(500, "INTERNAL_ERROR", "合同元信息提取失败")

    # 2. 用用户选择的类型覆盖
    meta["contract_type"] = contract_type
    now = _now_iso()
    meta["review_time"] = now
    meta["created_at"] = now
    meta["updated_at"] = now

    # 3. 写入 contracts 表
    try:
        repository = get_contract_repository()
        contract_id = repository.upsert_by_file_id(meta)
        contract = repository.find_by_id(contract_id)
    except RuntimeError:
        return error_response(500, "INTERNAL_ERROR", "合同记录创建失败")

    if contract is None:
        return error_response(500, "INTERNAL_ERROR", "合同记录创建失败")

    # 4. 创建审核任务（复用现有逻辑）
    try:
        points = [
            point for point_id in point_ids
            if (point := audit_point_repository.find_by_id(point_id)) is not None
        ]
        if not points:
            return error_response(422, "POINTS_REQUIRED", "未找到可用审核点")

        task_id = audit_result_store.create_task(contract_id, role)
        original_blocks = _original_text_for_contract(contract, [])
        items = _build_audit_items(points, original_blocks)
        stats = _build_dimension_stats(items)
        audit_result_store.insert_results(task_id, items)
        audit_result_store.insert_stats(task_id, stats)
        major_count = sum(1 for item in items if item["risk_level"] == 1)
        general_count = len(items) - major_count
        audit_result_store.complete_task(task_id, total=len(items), major=major_count, general=general_count)
        risk, risk_level = _risk_summary(major_count, general_count)
        repository.update_risk_cache(contract_id, risk=risk, risk_count=len(items), risk_level=risk_level)
    except RuntimeError:
        return error_response(500, "INTERNAL_ERROR", "审核执行失败")

    return flat_object({"id": str(contract_id), "status": "done"})
```

### 1.4 文件清理端点

**文件**：`api/files.py`

新增 `DELETE /api/files/{file_id}/artifacts`：

```python
@router.delete("/{file_id}/artifacts")
async def delete_upload_artifacts(file_id: str) -> JSONResponse:
    """删除上传文件 + 解析目录。用于用户取消上传流程。"""
    settings = get_settings()

    # 删除上传文件及其元数据
    if settings.upload_root.exists():
        for path in settings.upload_root.rglob(f"{file_id}.*"):
            if path.is_file():
                try:
                    path.unlink(missing_ok=True)
                except OSError:
                    pass

    # 删除解析目录
    shutil.rmtree(settings.parsing_root / file_id, ignore_errors=True)

    return ok_deleted()
```

需在文件顶部增加 `import shutil`。

## 二、前端改造

### 2.1 `contract.service.js` — 新增清理函数 + 修改 audit 调用

```javascript
// 新增：清理上传文件
export async function cleanupUpload(fileId) {
  if (USE_MOCK) return delay({ ok: true })
  const { data } = await http.delete(`/files/${fileId}/artifacts`)
  return data
}

// 修改：startAudit 发送 fileId + contractType
export async function startAudit(payload) {
  if (USE_MOCK) {
    await delay(null, 500)
    return { id: nextId('c'), status: 'done' }
  }
  const { data } = await http.post('/contracts/audit', payload)
  return data
}
```

请求体从 `{ role, points }` 变为 `{ fileId, role, points, contractType }`。

## 三、数据流对比

### 改造前

```
上传 → 写 contracts(id=5) → 返回 {id:"5", fileId, ...}
     → frontend stores contractId="5"
     → startAudit("5", {role, points}) → POST /contracts/5/audit
```

### 改造后

```
上传 → 只保存文件+解析 → 返回 {fileId, name, detectedType, ...}
     → frontend stores fileId（无 contractId）
     
返回/重传 → DELETE /files/{fileId}/artifacts → 清理磁盘
           → reset store → 回到 Step1

开始分析 → POST /contracts/audit {fileId, role, points, contractType}
         → 后端写 contracts(id=5) → 创建 audit_task → 返回 {id:"5", status}
         → frontend stores contractId="5" → 跳转结果页
```

  → 结构编辑 → 返回上传 → check 文件/解析目录已删除

[重新上传清理]
